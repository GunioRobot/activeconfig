  def _fire_on_load(name)
    callbacks = 
      (@on_load['ANY'] || EMPTY_ARRAY) + 
      (@on_load[name] || EMPTY_ARRAY)
    callbacks.uniq!
    STDERR.puts "_fire_on_load(#{name.inspect}): callbacks = #{callbacks.inspect}" if @verbose && ! callbacks.empty?
    callbacks.each do | cb |
      cb.call()
    end
  end
