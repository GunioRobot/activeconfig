  def _config_path_ary
    @config_path_ary ||=
      begin
        path_sep = (_config_path =~ /;/) ? /;/ : /:/ # Make Wesha happy
        path = _config_path.split(path_sep).reject{ | x | x.empty? }
        path = 
          path.collect! do | x | 
            x.freeze
          end
        path.freeze
        path
      end
  end
