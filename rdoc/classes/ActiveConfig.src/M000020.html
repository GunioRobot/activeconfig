  def _make_indifferent_and_freeze(x)
    return x if x.frozen?
    case x
    when HashConfig
        x.each_pair do | k, v |
          x[k] = _make_indifferent_and_freeze(v)
        end
    when Hash
      x = HashConfig.new.merge!(x)
      x.each_pair do | k, v |
        x[k] = _make_indifferent_and_freeze(v)
      end
      # STDERR.puts "x = #{x.inspect}:#{x.class}"
    when Array
      x.collect!  do | v |
        _make_indifferent_and_freeze(v)
      end
    end
    x.freeze
  end
