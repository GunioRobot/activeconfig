    def method_missing method, val=nil 
      super if method.to_s=~/^_/
      if method.to_s=~/^(.*)=$/
        @parent._flush_cache
        return @symbols[$1]=val
      end      
      ret=@symbols[method]
      if ret 
        return ret.call(@symbols) if ret.respond_to?(:call)
        return ret
      end
      super
    end
